"""Cost function: - sum(y_n.log(y^h_n) + (1-y_n).log(1-y^h_n))"""
def cal_cost(y, y_hat):
    m = y.shape[0]
    left = np.multiply(y, np.log(y_hat))
    right = np.multiply(np.subtract(1, y), np.log(np.subtract(1, y_hat)))
    total = np.sum(np.add(left, right), axis=0)
    return -total / m

"""Calculate delta w'"""
cal_delta_w = lambda X, y, y_hat: np.divide(np.dot(np.subtract(y_hat, y).T, X), X.shape[0])

"""Gradient descent"""
while l_count < l_max:
    l_count += 1
    
    z_curr = np.dot(X, w_arr[-1].T)
    y_hat = sigmoid_func(z_curr)

    e_curr = cal_cost(y, y_hat)
    e_arr.append(e_curr)
    if e_arr[-1] < accepted_value:
        break
    
    w_new = w_arr[-1] - lear_rate * cal_delta_w(X, y, y_hat)
    w_arr.append(w_new)
