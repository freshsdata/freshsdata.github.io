"""Assumed function"""
poly_func = lambda x: 3*np.sin(2 * np.pi * x)

def recover_func(z, m, w):
    """
    Equation of recoverd function
    y = w_0*x_0^0 + w_1*x_1^1 + ... + w_m*x_m^m
    """
    y_hat = np.zeros(z.shape)
    for i in range(0, m + 1):
        z_power = np.power(z, i)
        y_hat = y_hat + (w[i, 0] * z_power)
    return y_hat

"""Generate X from -pi to pi with N samples"""
gen_X = lambda N: np.random.uniform(low=-.5, high=.5, size=(N,1))

def gen_X_bar(X, M):
    """
    Generate X_bar for polynomial equation from vector X
    """
    X_bar = np.ones((X.shape[0], 1))
    for i in range(1, M + 1):
        x_power = np.power(X, i)
        X_bar = np.concatenate((X_bar, x_power), axis = 1)
    return X_bar

"""Generate label t"""
gen_t = lambda X: poly_func(X) + np.random.normal(0, .1, X.shape)

"""Calculate w from equation"""
cal_w = lambda X_bar, t: np.dot(np.linalg.pinv(np.dot(X_bar.T, X_bar)), np.dot(X_bar.T, t))

def plot_figure(X, t, M, w):
    """
    Plot figure
    """

    """Plot input data"""
    x_plt = np.arange(-.5, .5, .01)
    plt.figure(figsize=(12, 6))
    plt.plot(X, t, 'ro')
    
    """Plot original function"""
    plt.plot(x_plt, poly_func(x_plt), 'b')

    """Plot recoverd function by equation"""
    y_plot = recover_func(x_plt, M, w)
    plt.plot(x_plt, y_plot, 'y')

    """Show figure"""
    plt.ylabel('y axis')
    plt.xlabel('x axis')
    plt.suptitle('Polynomial Regression', fontsize=24)
    plt.grid(True)
    plt.legend(('Data', r'$h(x)$', r'$y(x, w)$'),
           shadow=True, loc=0, handlelength=1.5, fontsize=20)
    plt.show()

"""Function to calculate 'trung bình tổng bình phương sai số'"""
cal_equa = lambda X, t, w: np.sum(np.square(np.subtract(t, np.dot(X, w.T))), axis=0) / X.shape[0]

"""Function to calculate 'đạo hàm của hàm trung bình tổng bình phương sai số'"""
cal_grad = lambda X, t, w: np.dot(np.subtract(np.dot(X, w.T), t).T, X) / X.shape[0]

def GD_func(X, t, lear_rate=.01, vali_delta=1e-6, w_init_RMSE=10, l_number=math.inf):
    """
    Implementation of Gradient Descent function is here
    """
    
    l_count = 0
    w_zero = gen_w((1, X.shape[1]), w_init_RMSE)
    W_arr = [w_zero]
    E_arr = [np.square(np.amax(t) - np.amin(t))]
    
    while l_count < l_number:
        l_count += 1

        e_curr = cal_equa(X, t, W_arr[-1])
        E_arr.append(e_curr)

        if abs(E_arr[-1] - E_arr[-2]) <= vali_delta:
            break
        else:
            w_new = W_arr[-1] - lear_rate * cal_grad(X, t, W_arr[-1])
            W_arr.append(w_new)
            
    return W_arr[-1], l_count, E_arr[:]
